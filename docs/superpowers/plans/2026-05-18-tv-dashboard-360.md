- [ ] **Paso 1.2 — Insertar el bloque `planillas_activas` después de `unidades_pendientes` (línea ~1579)**

Insertar el siguiente bloque entre el cierre de `$stats['unidades_pendientes']` y `// Faltantes Críticos`:

```php
        // Planillas activas en vivo (agrupadas por planilla_numero)
        $ordenesActivas = OrdenPicking::where('empresa_id', $empresaId)
            ->where('sucursal_id', $user->sucursal_id)
            ->whereBetween('created_at', [$ini, $fin])
            ->whereIn('estado', ['Pendiente', 'EnProceso'])
            ->when($params['auxiliar_id'] ?? null, fn($q, $a) => $q->where('auxiliar_id', $a))
            ->when($params['ruta'] ?? null, fn($q, $ruta) => $q->where('area_comercial', $ruta))
            ->withCount([
                'detalles as total_lineas',
                'detalles as lineas_completadas' => fn($q) => $q->whereIn('estado', ['Completado', 'Faltante']),
            ])
            ->with([
                'detalles' => fn($q) => $q->select('id', 'orden_picking_id', 'auxiliar_id', 'estado')
                                           ->with('auxiliar:id,nombre'),
            ])
            ->get();

        $stats['planillas_activas'] = $ordenesActivas
            ->groupBy(fn($o) => $o->planilla_numero ?? $o->numero_orden)
            ->map(function ($orders, $planillaKey) {
                $first       = $orders->first();
                $totalLineas = $orders->sum('total_lineas');
                $lineasComp  = $orders->sum('lineas_completadas');
                $estado      = $orders->contains('estado', 'EnProceso') ? 'EnProceso' : 'Pendiente';
                $horaInicio  = $orders->whereNotNull('hora_inicio')
                                      ->where('hora_inicio', '!=', '00:00:00')
                                      ->min('hora_inicio');
                $auxiliares  = $orders->flatMap(fn($o) => $o->detalles->pluck('auxiliar.nombre'))
                                      ->filter()->unique()->values();
                $tieneFalt   = $orders->flatMap(fn($o) => $o->detalles)
                                      ->contains('estado', 'Faltante');

                return [
                    'planilla_numero'    => $planillaKey,
                    'estado'             => $estado,
                    'ruta'               => $first->area_comercial,
                    'hora_inicio'        => $horaInicio,
                    'total_lineas'       => $totalLineas,
                    'lineas_completadas' => $lineasComp,
                    'auxiliares'         => $auxiliares,
                    'tiene_faltante'     => $tieneFalt,
                ];
            })
            ->sortBy(fn($p) => $p['estado'] === 'EnProceso' ? 0 : 1)
            ->values();
```

Reemplazar el bloque completo (líneas 1604–1618) con:

```php
        // Ranking de Auxiliares (Pedidos, Líneas, Unidades, Tiempo Promedio)
        $stats['ranking_auxiliares'] = Capsule::table('personal as aux')
            ->join('picking_detalles as d', 'aux.id', '=', 'd.auxiliar_id')
            ->join('orden_pickings as o', 'd.orden_picking_id', '=', 'o.id')
            ->where('o.empresa_id', $empresaId)
            ->where('o.sucursal_id', $user->sucursal_id)
            ->whereBetween('o.created_at', [$ini, $fin])
            ->when($params['auxiliar_id'] ?? null, fn($q, $a) => $q->where('aux.id', $a))
            ->select(
                'aux.id',
                'aux.nombre',
                Capsule::raw('COUNT(DISTINCT o.id) as pedidos'),
                Capsule::raw('COUNT(d.id) as lineas'),
                Capsule::raw('SUM(d.cantidad_pickeada) as unidades'),
                Capsule::raw(
                    'ROUND(AVG(CASE WHEN o.hora_fin IS NOT NULL AND o.hora_inicio IS NOT NULL
                        AND o.hora_inicio != \'00:00:00\'
                        THEN TIME_TO_SEC(TIMEDIFF(o.hora_fin, o.hora_inicio)) / 60
                        ELSE NULL END), 1) as avg_minutos'
                )
            )
            ->groupBy('aux.id', 'aux.nombre')
            ->orderByDesc('lineas')
            ->limit(10)
            ->get();
```

// ── BASE URL ─────────────────────────────────────────────────────────────────
const BASE = (() => {
  const p = window.location.pathname;
  const m = p.match(/^(\/[^/]+\/public)\//);
  return m ? m[1] + '/api' : window.location.origin + '/api';
})();

// ── API ──────────────────────────────────────────────────────────────────────
async function api(path) {
  const r = await fetch(BASE + path, {
    headers: { Authorization: 'Bearer ' + token, Accept: 'application/json' }
  });
  if (!r.ok) throw new Error('HTTP ' + r.status + ' en ' + path);
  const j = await r.json();
  return j.data ?? j;
}

function tick() {

  setText('tv-clock', n.toLocaleTimeString('es-CO'));
  setText('tv-date',  n.toLocaleDateString('es-CO',
    { weekday:'long', year:'numeric', month:'long', day:'numeric' }));
}
setInterval(tick, 1000); tick();

// ── HELPERS ──────────────────────────────────────────────────────────────────
function getToday() {
  const d = new Date();
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}
function fmtDate(d) {
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}
function setText(id, val) {
  const el = document.getElementById(id);
  if (el) el.textContent = val;
}
function fmt(n) {
  return new Intl.NumberFormat('es-CO').format(Math.round(Number(n) || 0));
}
function esc(s) {
  return String(s ?? '').replace(/[&<>"']/g,
    c => ({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;' }[c]));
}
function showError(msg) {
  const el = document.getElementById('tv-error');
  if (el) { el.textContent = msg; el.style.display = 'block'; }
}
function hideError() {

  if (el) el.style.display = 'none';
}

// Calcula tiempo transcurrido desde hora "HH:MM:SS" combinando con la fecha local de hoy
function elapsedTime(horaInicio) {
  if (!horaInicio || horaInicio === '00:00:00') return null;
  const d = new Date();
  const [h, m, s] = horaInicio.split(':').map(Number);
  const start = new Date(d.getFullYear(), d.getMonth(), d.getDate(), h, m, s);
  const diff = Math.floor((Date.now() - start.getTime()) / 1000);
  if (diff < 0) return null;
  const hrs  = Math.floor(diff / 3600);
  const mins = Math.floor((diff % 3600) / 60);
  return hrs > 0 ? `${hrs}h ${mins}m` : `${mins}m`;
}

// ── COUNTDOWN ────────────────────────────────────────────────────────────────
function startCountdown() {
  if (_countTimer) clearInterval(_countTimer);
  _countDown = REFRESH_INTERVAL;
  _countTimer = setInterval(() => {
    _countDown--;
    setText('next-refresh', `Próxima actualización: ${_countDown}s`);
    if (_countDown <= 0) { clearInterval(_countTimer); refresh(); }
  }, 1000);
}

// ── FILTROS ──────────────────────────────────────────────────────────────────
function setupFilters() {
  const today = getToday();
  filters.fecha_inicio = today;
  filters.fecha_fin    = today;
  document.getElementById('f-desde').value = today;
  document.getElementById('f-hasta').value = today;
  document.querySelector('[data-preset="hoy"]').classList.add('active');

  document.querySelectorAll('[data-preset]').forEach(btn => {
    btn.addEventListener('click', () => {
      document.querySelectorAll('[data-preset]').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      const today = getToday();
      const preset = btn.dataset.preset;
      if (preset === 'hoy') {
        filters.fecha_inicio = today; filters.fecha_fin = today;
      } else if (preset === 'semana') {
        const d = new Date(); d.setDate(d.getDate() - ((d.getDay() + 6) % 7));
        filters.fecha_inicio = fmtDate(d); filters.fecha_fin = today;
      } else if (preset === 'mes') {
        const d = new Date(); d.setDate(1);
        filters.fecha_inicio = fmtDate(d); filters.fecha_fin = today;
      } else {
        return; // custom: wait for date input
      }
      document.getElementById('f-desde').value = filters.fecha_inicio;
      document.getElementById('f-hasta').value = filters.fecha_fin;
      refresh();
    });
  });

  document.getElementById('f-desde').addEventListener('change', e => {
    filters.fecha_inicio = e.target.value;
    document.querySelectorAll('[data-preset]').forEach(b => b.classList.remove('active'));
    refresh();
  });
  document.getElementById('f-hasta').addEventListener('change', e => {
    filters.fecha_fin = e.target.value;
    document.querySelectorAll('[data-preset]').forEach(b => b.classList.remove('active'));
    refresh();
  });
  document.getElementById('f-auxiliar').addEventListener('change', e => {
    filters.auxiliar_id = e.target.value; refresh();
  });
  document.getElementById('f-ruta').addEventListener('change', e => {
    filters.ruta = e.target.value; refresh();
  });
  document.getElementById('btn-refresh').addEventListener('click', () => refresh());
}

function buildQS() {
  const p = new URLSearchParams({
    fecha_inicio: filters.fecha_inicio,
    fecha_fin:    filters.fecha_fin,
  });
  if (filters.auxiliar_id) p.set('auxiliar_id', filters.auxiliar_id);
  if (filters.ruta)        p.set('ruta', filters.ruta);
  return p.toString();
}

function populateSelects(d) {
  // Auxiliar select — poblar desde ranking (tiene id + nombre)
  const auxSel = document.getElementById('f-auxiliar');
  const prevAux = auxSel.value;
  auxSel.innerHTML = '<option value="">Todos</option>';
  (d.ranking_auxiliares || []).forEach(r => {
    if (!r.id) return;
    const o = document.createElement('option');
    o.value = r.id; o.textContent = r.nombre;
    if (String(r.id) === prevAux) o.selected = true;
    auxSel.appendChild(o);
  });

  // Ruta select — poblar desde planillas_activas (valores únicos)
  const rutaSel = document.getElementById('f-ruta');
  const prevRuta = rutaSel.value;
  const rutas = [...new Set((d.planillas_activas || []).map(p => p.ruta).filter(Boolean))].sort();
  rutaSel.innerHTML = '<option value="">Todas</option>';
  rutas.forEach(r => {
    const o = document.createElement('option');
    o.value = r; o.textContent = r;
    if (r === prevRuta) o.selected = true;
    rutaSel.appendChild(o);
  });
}

// ── KPIs ─────────────────────────────────────────────────────────────────────
function renderKPIs(d) {
  const activas    = (d.pendientes || 0) + (d.en_proceso || 0);
  const totalLin   = d.total_lineas_activas || 0;
  const pendLin    = d.lineas_pendientes    || 0;
  const compLin    = totalLin - pendLin;
  const pct        = totalLin > 0 ? Math.round(compLin / totalLin * 100) : 0;
  const pctColor   = pct >= 80 ? 'var(--green)' : pct >= 50 ? 'var(--yellow)' : 'var(--red)';

  setText('k-activas',    activas);
  setText('k-completadas', d.completadas || 0);

  const pctEl = document.getElementById('k-pct');
  if (pctEl) { pctEl.textContent = pct + '%'; pctEl.style.color = pctColor; }
  setText('k-pct-sub',     `${fmt(compLin)} de ${fmt(totalLin)} líneas`);
  setText('k-lineas-pend', fmt(pendLin));
  setText('k-uds-pend',    fmt(d.unidades_pendientes || 0));
  setText('k-faltantes',   (d.alertas_faltantes || []).length);
}

// ── GRÁFICAS ─────────────────────────────────────────────────────────────────
const COLORS = ['#58a6ff','#3fb950','#d29922','#bc8cff','#ffa657','#79c0ff','#f0883e','#56d364','#db61a2','#e3b341'];

function buildChart(id, wrapId, labels, data, unitLabel) {
  if (charts[id]) { charts[id].destroy(); delete charts[id]; }
  const wrap = document.getElementById(wrapId);
  if (!wrap) return;

  if (!labels.length) {
    wrap.innerHTML = '<div class="no-data">Sin datos para el período seleccionado</div>';
    return;
  }

  // Restaurar canvas si fue reemplazado por el mensaje no-data
  if (!wrap.querySelector('canvas')) {
    wrap.innerHTML = `<canvas id="${id}"></canvas>`;
  }

  const ctx = document.getElementById(id);
  if (!ctx) return;

  charts[id] = new Chart(ctx, {
    type: 'bar',
    data: {
      labels,
      datasets: [{
        label: unitLabel, data,
        backgroundColor: labels.map((_,i) => COLORS[i % COLORS.length] + 'bb'),
        borderColor:     labels.map((_,i) => COLORS[i % COLORS.length]),
        borderWidth: 2, borderRadius: 4
      }]
    },
    options: {
      indexAxis: 'y',
      responsive: true, maintainAspectRatio: false,
      animation: { duration: 500 },
      plugins: {
        legend: { display: false },
        tooltip: { callbacks: { label: c => ` ${c.parsed.x} ${unitLabel.toLowerCase()}` } }
      },
      scales: {
        x: { ticks:{ color:'#8b949e', font:{size:9} }, grid:{ color:'#21262d' }, beginAtZero:true },
        y: { ticks:{ color:'#e6edf3', font:{size:10,weight:'bold'} }, grid:{ display:false } }
      }
    }
  });
}

function renderCharts(ranking) {
  const sorted = [...(ranking || [])].sort((a, b) => (b.lineas || 0) - (a.lineas || 0));
  const names  = sorted.map(r => r.nombre || 'N/A');

  buildChart('chart-lineas',   'wrap-lineas',  names, sorted.map(r => r.lineas  || 0),  'líneas');

  const withTime  = sorted.filter(r => r.avg_minutos > 0);
  const timeNames = withTime.map(r => r.nombre || 'N/A');
  buildChart('chart-tiempos', 'wrap-tiempos', timeNames, withTime.map(r => Math.round(r.avg_minutos || 0)), 'min promedio');
}

// ── LOAD / REFRESH ───────────────────────────────────────────────────────────
async function loadPicking() {
  const d = await api(`/picking/dashboard?${buildQS()}`);
  populateSelects(d);
  renderKPIs(d);
  renderPlanillasTable(d.planillas_activas || []);
  renderCharts(d.ranking_auxiliares || []);
  renderAlertas(d.alertas_faltantes  || []);
}

async function refresh() {
  const spin = document.getElementById('refresh-spin');
  if (spin) spin.style.display = 'inline-block';
  try {
    await loadPicking();
    hideError();
  } catch(e) {
    showError('Error al cargar datos: ' + e.message);
  } finally {
    if (spin) spin.style.display = 'none';
    startCountdown();
  }
}

async function init() {
  _initialized = true;

    return;
  }
  await refresh();
}
